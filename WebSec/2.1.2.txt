CREATE TABLE users ( id INT UNSIGNED AUTO_INCREMENT NOT NULL, username VARCHAR(32) NOT NULL, salt VARCHAR(64) NOT NULL, passwordhash VARCHAR(64) NOT NULL, PRIMARY KEY (id));
CREATE UNIQUE INDEX username ON users(username);
CREATE TABLE history ( id INT UNSIGNED AUTO_INCREMENT NOT NULL, user_id INT UNSIGNED NOT NULL, query VARCHAR(2048) NOT NULL, PRIMARY KEY (id));
CREATE INDEX user_id ON history (user_id);
